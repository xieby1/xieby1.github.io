int main(void)
{
   asm( "mov a, %eax\n\t"
        "mov b, %ebx\n\t"
        "add %ebx, %eax\n\t"
        "mov %eax, result");
   printf("the result is %d\n", result);

}

#include <stdio.h>
int a = 10;
int b = 20;
int result;
int main(void)
{
   asm( "mov a, %%eax\n\t"
        "mov b, %%ebx\n\t"
        "add %%ebx, %%eax\n\t"
        "mov %%eax, result"
        : /* no outputs*/
        : /* no inputs */
        : "eax", "ebx");
   printf("the result is %d\n", result);
   return 0;
